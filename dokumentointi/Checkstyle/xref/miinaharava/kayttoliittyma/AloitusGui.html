1   package miinaharava.kayttoliittyma;
2   
3   import java.awt.*;
4   import javax.swing.*;
5   import miinaharava.kayttoliittyma.kuuntelijat.AloitusGuinKuuntelija;
6   
7   public class AloitusGui implements Runnable {
8   
9       private JDialog dialog;
10      private JComboBox<String> valikko;
11      private JTextField nimimerkki;
12      private JLabel nimimerkkiKentanOtsikko;
13  
14      /**
15       * Luo aloitusikkunan Miinaharava pelin graafiseen käyttöliittymään.
16       */
17      public AloitusGui() {
18          valikko = new JComboBox<String>();
19      }
20  
21      @Override
22      public void run() {
23          dialog = new JDialog();
24  
25          dialog.setDefaultCloseOperation(WindowConstants.DISPOSE_ON_CLOSE);
26  
27          luoKomponentit(dialog.getContentPane());
28  
29          dialog.setResizable(false);
30          dialog.pack();
31          dialog.setLocationRelativeTo(dialog);
32          dialog.setVisible(true);
33      }
34  
35      /**
36       * Luo käyttöliittymäkomponentit aloitusnäkymään.
37       *
38       * @param container Container johon käyttöliittymäkomponentit sijoitetaan.
39       */
40      public void luoKomponentit(Container container) {
41          container.setLayout(new BorderLayout());
42          container.add(teksti(), BorderLayout.NORTH);
43          container.add(luoVaihtoehdot(), BorderLayout.CENTER);
44          container.add(luoNimimerkkiKentta(), BorderLayout.SOUTH);
45          container.add(luoNapit(), BorderLayout.EAST);
46  
47      }
48  
49      /**
50       * Luo napin jolla peli aloitetaan.
51       *
52       * @return Palauttaa JPanel komponentin.
53       */
54      private JPanel luoNapit() {
55          JPanel paneeli = new JPanel();
56          JButton aloita = new JButton("Aloita");
57          JButton tuloslista = new JButton("tuloslista");
58          aloita.addActionListener(new AloitusGuinKuuntelija(dialog, valikko, nimimerkki, nimimerkkiKentanOtsikko));
59          tuloslista.addActionListener(new AloitusGuinKuuntelija(dialog, valikko, nimimerkki, nimimerkkiKentanOtsikko));
60          paneeli.add(aloita);
61          paneeli.add(tuloslista);
62          return paneeli;
63      }
64  
65      /**
66       * Luo tekstin aloitusnäkymään.
67       *
68       * @return Paulauttaa JLabel komponentin.
69       */
70      public JLabel teksti() {
71          JLabel teksti = new JLabel("Valitse vaikeusaste");
72          teksti.setVerticalAlignment(SwingConstants.CENTER);
73          teksti.setHorizontalAlignment(SwingConstants.CENTER);
74          teksti.setFont(new Font("normal", 0, 20));
75          return teksti;
76      }
77  
78      /**
79       * Luo JRadioButtonit eri vaikeusvaihtoehdoille.
80       *
81       * @return Palauttaa JPanel komponentin.
82       */
83      public JPanel luoVaihtoehdot() {
84          JPanel paneeli = new JPanel();
85          valikko = new JComboBox<>();
86          String[] vaihtoehdot = {"Helppo (8x8, 10 miinaa)", "Haastava (16x16, 40 miinaa)", "Vaikea (16x30, 99 miinaa)", "Custom"};
87          for (int i = 0; i < vaihtoehdot.length; i++) {
88              valikko.addItem(vaihtoehdot[i]);
89          }
90          paneeli.add(valikko);
91          return paneeli;
92      }
93  
94      /**
95       * Luo kentän johon pelaaja voi kirjoittaa nimimerkkinsä.
96       *
97       * @return Palauttaa JPanel komponentin
98       */
99      public JPanel luoNimimerkkiKentta() {
100         JPanel paneeli = new JPanel(new BorderLayout());
101         nimimerkkiKentanOtsikko = new JLabel("Anna nimimerkki");
102         nimimerkkiKentanOtsikko.setFont(new Font("Normal", Font.PLAIN, 14));
103         nimimerkki = new JTextField();
104         paneeli.add(nimimerkkiKentanOtsikko, BorderLayout.NORTH);
105         paneeli.add(nimimerkki, BorderLayout.CENTER);
106         return paneeli;
107 
108     }
109 
110 }
