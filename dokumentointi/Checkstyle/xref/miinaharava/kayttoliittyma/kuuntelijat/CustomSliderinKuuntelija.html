1   package miinaharava.kayttoliittyma.kuuntelijat;
2   
3   import javax.swing.*;
4   import javax.swing.event.*;
5   import miinaharava.kayttoliittyma.CustomPelinLuontiIkkuna;
6   
7   public class CustomSliderinKuuntelija implements ChangeListener {
8   
9       private final JLabel teksti;
10      private final CustomPelinLuontiIkkuna frame;
11  
12      /**
13       * Luo ChangeListenerin CustomPelinLuontiIkkunan slidereille.
14       *
15       * @param teksti Päivitettävä teksti
16       * @param frame Päivitettävä JFrame
17       */
18      public CustomSliderinKuuntelija(JLabel teksti, CustomPelinLuontiIkkuna frame) {
19          this.teksti = teksti;
20          this.frame = frame;
21      }
22  
23      @Override
24      public void stateChanged(ChangeEvent e) {
25          JSlider muutettu = (JSlider) e.getSource();
26          if (teksti.getText().contains("Korkeus")) {
27              teksti.setText("Korkeus: " + muutettu.getValue());
28              frame.paivitaMiinat();
29          }
30          if (teksti.getText().contains("Leveys")) {
31              teksti.setText("Leveys: " + muutettu.getValue());
32              frame.paivitaMiinat();
33          }
34          if (teksti.getText().contains("Miinat")) {
35              teksti.setText("Miinat: " + muutettu.getValue());
36              frame.paivitaMiinat();
37          }
38      }
39  
40  }
