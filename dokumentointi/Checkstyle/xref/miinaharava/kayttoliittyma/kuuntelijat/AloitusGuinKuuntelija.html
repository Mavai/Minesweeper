3   import java.awt.event.*;
4   import javax.swing.*;
5   import miinaharava.kayttoliittyma.CustomPelinLuontiIkkuna;
6   import miinaharava.kayttoliittyma.PelikenttaGui;
7   import miinaharava.kayttoliittyma.Tulosnakyma;
8   import miinaharava.logiikka.Miinaharava;
9   import miinaharava.logiikka.Vaikeusaste;
10  
11  public class AloitusGuinKuuntelija implements ActionListener {
12  
13      private JDialog dialog;
14      private JComboBox<String> valikko;
15      private JTextField nimimerkki;
16      private JLabel tekstikentta;
17  
18      /**
19       * Aloittaa pelin riippuen siitä mikä vaihtoehto aloitusikkunassa valittiin.
20       *
21       * @param dialog Aloitusikkuna
22       * @param valikko JComboBox
23       * @param nimimerkki Tekstikenttä johon pelaaja syöttää nimimerkkinsä
24       * @param tekstikentta Nimimerkkikentän otsikko jota päivitetään
25       */
26      public AloitusGuinKuuntelija(JDialog dialog, JComboBox valikko, JTextField nimimerkki, JLabel tekstikentta) {
27          this.dialog = dialog;
28          this.valikko = valikko;
29          this.nimimerkki = nimimerkki;
30          this.tekstikentta = tekstikentta;
31      }
32  
33      @Override
34      public void actionPerformed(ActionEvent e) {
35          JButton painettu = (JButton) e.getSource();
36          if (painettu.getText().equals("Aloita")) {
37              String valinta = (String) valikko.getSelectedItem();
38              Miinaharava peli = null;
39              if (valinta.contains("Helppo")) {
40                  peli = new Miinaharava(Vaikeusaste.HELPPO);
41              } else if (valinta.contains("Vaikea")) {
42                  peli = new Miinaharava(Vaikeusaste.VAIKEA);
43              } else if (valinta.contains("Haastava")) {
44                  peli = new Miinaharava(Vaikeusaste.HAASTAVA);
45              } else if (valinta.contains("Custom")) {
46                  CustomPelinLuontiIkkuna uusiIkkuna = new CustomPelinLuontiIkkuna();
47                  dialog.dispose();
48                  return;
49              }
50              if (!asetaPelaaja(peli)) {
51                  return;
52              }
53              PelikenttaGui uusiPeli = new PelikenttaGui(peli);
54              dialog.dispose();
55          } else {
56              Tulosnakyma tulokset = new Tulosnakyma();
57          }
58      }
59  
60      /**
61       * Asettaa pelaajan syottaman nimimerkin jos se ei sisalla kiellettyja
62       * merkkeja.
63       *
64       * @param peli Käynnistettänä peli.
65       * @return Palauttaa true jos nimimerkki kelpaa, muuten false.
66       */
67      public boolean asetaPelaaja(Miinaharava peli) {
68          if (nimimerkki.getText().contains("-")) {
69              tekstikentta.setText("Anna nimimerkki     Nimi ei saa sisältää merkkiä '-'");
70              return false;
71          }
72          if (!nimimerkki.getText().equals("")) {
73              peli.setPelaaja(nimimerkki.getText());
74          }
75          return true;
76      }
77  
78  }
