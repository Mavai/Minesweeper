3   import miinaharava.kayttoliittyma.kuuntelijat.ValikonKuuntelija;
4   import miinaharava.kayttoliittyma.kuuntelijat.RuudunAvaaja;
5   import miinaharava.kayttoliittyma.kuuntelijat.KellonKuuntelija;
6   import java.awt.*;
7   import javax.swing.*;
8   import miinaharava.domain.Pelialusta;
9   import miinaharava.logiikka.Miinaharava;
10  
11  public class PelikenttaGui extends JFrame {
12  
13      private Pelialusta pelialusta;
14      private int leveys;
15      private JButton[][] ruudut;
16      private Miinaharava peli;
17      private JLabel kulunutAika;
18      private JLabel miinojenLkm;
19      private Timer kello;
20  
21      /**
22       * Luo graafisen käyttöliittymän pelille.
23       *
24       * @param peli Peli, joka käytttää käyttöliittymää.
25       */
26      public PelikenttaGui(Miinaharava peli) {
27          super("Miinaharava");
28          setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
29  
30          this.peli = peli;
31          this.pelialusta = peli.getPelialusta();
32          this.leveys = peli.getPelialusta().getLeveys();
33          ruudut = new JButton[leveys][pelialusta.getKorkeus()];
34  
35          this.kulunutAika = new JLabel("0  :  0", JLabel.CENTER);
36          this.miinojenLkm = new JLabel("Miinoja jäljellä  " + peli.getMiinojaJaljella(), JLabel.CENTER);
37          this.kello = new Timer(1000, new KellonKuuntelija(kulunutAika));
38          luoKomponentit(this.getContentPane());
39          setResizable(false);
40          pack();
41          setLocationRelativeTo(this);
42          setVisible(true);
43          kello.start();
44      }
45  
46      private void luoKomponentit(Container container) {
47          container.setLayout(new BorderLayout());
48          container.add(valikko(), BorderLayout.NORTH);
49          JPanel ruudukko = luoRuudukko();
50          container.add(ruudukko, BorderLayout.CENTER);
51          JPanel kierrokset = luoKierrostenJaMerkkienLaskija();
52          kierrokset.setPreferredSize(new Dimension(30 * leveys, 30));
53          container.add(kierrokset, BorderLayout.SOUTH);
54      }
55  
56      /**
57       * Luo käyttöliittymään JPanel olion joka sisältää kierroslakurin ja
58       * merkkausten määrän.
59       *
60       * @return JPanel käyttöliittymäkomponentti
61       */
62      private JPanel luoKierrostenJaMerkkienLaskija() {
63          JPanel kierrokset = new JPanel(new GridLayout(1, 2));
64          miinojenLkm.setFont(new Font("Normal", Font.BOLD, 12));
65          kulunutAika.setFont(new Font("Normal", Font.BOLD, 12));
66          miinojenLkm.setBorder(BorderFactory.createLineBorder(Color.BLACK));
67          kulunutAika.setBorder(BorderFactory.createLineBorder(Color.BLACK));
68          kierrokset.add(kulunutAika);
69          kierrokset.add(miinojenLkm);
70          return kierrokset;
71      }
72  
73      /**
74       * Luo JPanel olion Grid layoutilla, joka sisältää JButtoneita. Toimii
75       * pelialustana.
76       *
77       * @return JPanel ruudukko
78       */
79      public JPanel luoRuudukko() {
80          JPanel ruudukko = new JPanel(new GridLayout(pelialusta.getKorkeus(), leveys));
81          for (int i = 0; i < pelialusta.getKorkeus(); i++) {
82              for (int j = 0; j < pelialusta.getLeveys(); j++) {
83                  JButton nappi = new JButton("", null);
84                  ruudut[j][i] = nappi;
85                  nappi.setPreferredSize(new Dimension(35, 35));
86                  nappi.setFont(new Font("Normal", Font.BOLD, 14));
87                  nappi.setMargin(new Insets(5, 5, 5, 5));
88                  nappi.addMouseListener(new RuudunAvaaja(j, i, peli, nappi, ruudut, miinojenLkm, kello, this));
89                  ruudukko.add(nappi);
90              }
91          }
92          return ruudukko;
93      }
94  /**
95   * Luo valikkopalkin peli-ikkunan yläreunaan.
96   * @return Palauttaa JMenuBar komponentin.
97   */
98      public JMenuBar valikko() {
99          JMenuBar valikko = new JMenuBar();
100         JMenu menu = new JMenu("Peli");
101         valikko.add(menu);
102         JMenuItem uusiPeli = new JMenuItem("Uusi peli");
103         uusiPeli.addActionListener(new ValikonKuuntelija(this, peli));
104         menu.add(uusiPeli);
105         menu.add(luoVaikeusasteenVaihtaja());
106         JMenuItem lopeta = new JMenuItem("Lopeta");
107         lopeta.addActionListener(new ValikonKuuntelija(this, peli));
108         menu.add(lopeta);
109         return valikko;
110     }
111 
112     private JMenu luoVaikeusasteenVaihtaja() {
113         JMenu vaihdaVaikeusastetta = new JMenu("Vaikeusaste");
114         JMenuItem helppo = new JMenuItem("Helppo");
115         JMenuItem haastava = new JMenuItem("Haastava");
116         JMenuItem vaikea = new JMenuItem("Vaikea");
117         JMenuItem demo = new JMenuItem("Demo");
118         helppo.addActionListener(new ValikonKuuntelija(this, peli));
119         haastava.addActionListener(new ValikonKuuntelija(this, peli));
120         vaikea.addActionListener(new ValikonKuuntelija(this, peli));
121         demo.addActionListener(new ValikonKuuntelija(this, peli));
122         vaihdaVaikeusastetta.add(helppo);
123         vaihdaVaikeusastetta.add(haastava);
124         vaihdaVaikeusastetta.add(vaikea);
125         vaihdaVaikeusastetta.add(demo);
126         return vaihdaVaikeusastetta;
127     }
128 
129     public Miinaharava getPeli() {
130         return peli;
131     }
132 
133     public JLabel getKulunutAika() {
134         return kulunutAika;
135     }
