1   package miinaharava.kayttoliittyma;
2   
3   import java.awt.Insets;
4   import java.awt.event.ActionEvent;
5   import java.awt.event.ActionListener;
6   import javax.swing.JButton;
7   import miinaharava.domain.Pelialusta;
8   import miinaharava.domain.Ruutu;
9   import miinaharava.logiikka.Miinaharava;
10  
11  /**
12   * RuudunAvaajaa kutsutaan kun painetaan pelialustalla olevaa JButtonia.
13   * Määrittelee painalluksen jälkeiset tapahtumat.
14   *
15   * @author markovai
16   */
17  public class RuudunAvaaja implements ActionListener {
18  
19      private int x;
20      private int y;
21      private Pelialusta alusta;
22      private Miinaharava peli;
23      private JButton nappi;
24      private JButton[][] ruudut;
25  
26      public RuudunAvaaja(int x, int y, Pelialusta alusta, JButton nappi, JButton[][] ruudut) {
27          this.alusta = alusta;
28          this.x = x;
29          this.y = y;
30          this.nappi = nappi;
31          this.peli = new Miinaharava(alusta, null);
32          this.ruudut = ruudut;
33      }
34  
35      @Override
36      public void actionPerformed(ActionEvent e) {
37          Ruutu avattava = alusta.getAlusta()[x][y];
38          peli.avaaRuutu(avattava);
39          nappi.setEnabled(false);
40          nappi.setText(avattava.toString());
41          nappi.setMargin(new Insets(5, 5, 5, 5));
42          if (avattava.getViereisetMiinat() == 0) {
43              avaaNollat(avattava);
44          }
45  
46      }
47  
48      public void avaaNollat(Ruutu avattava) {
49          for (Ruutu viereinenRuutu : avattava.viereisetRuudut(alusta.getAlusta(), x, y)) {
50              if (!viereinenRuutu.onKiinni()) {
51                  ruudut[viereinenRuutu.getX()][viereinenRuutu.getY()].doClick();
52              }
53          }
54  
55      }
56  }
